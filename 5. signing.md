1. intall gpg
	$ gpg --version

2. see keys
	$ gpg --list-keys

3. create key
	$ gpg --gen-key

4. get private key
	$ gpg --list-secret-keys --keyid-format LONG
	sec   ed25519/70D60C5485DFAE22 --> 70D60C5485DFAE22 is your private key

5. set sign for signing
	$ git config --global user.signingkey 70D60C5485DFAE22

6. see private key
	$ git config --global user.signingkey
	70D60C5485DFAE22

7. signing tag
	$ git tag -s v1.3 -m "sign last version of project"

8. verify sign tag
	$ git tag -v v1,3
	tag v1,3
	tagger HZALGO <hossein.zohurian.abotorabi@gmail.com> 1764348248 +0330

9. signing commit
	$ git commit -S -m 'commit with sign'
