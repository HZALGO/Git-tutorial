
1. found directory "~/desktop (master)"
	$ cd Git

2. intilization .git file
	$ git init

3. create new file
	$ touch example.xml

4. get git status(untracked file)
	$ git status

5. incloud file in .git --> No commits yet
	$ git add example.xml

6. commit with describ
	$ git commit -m 'create empty xml file'

7. with edit file and get status recive --> Changes not staged for commit
	$ git status

8. incloud file
	$ git add example.xml

9. commit
	$ git commit -m 'file was filled by massage'

10. history , (HEAD) mean last history  ,q for exit
	$ git log

11. changed on last commmit file('--' prevfile, '++' currfile,@lines to change)
	$ git diff HEAD

12. return from stage
	$ git reset filename.ex

13. change for staged file
	$ git diff --staged

14. return to curr file to prev file
	$ git checkout -- filename.ex
